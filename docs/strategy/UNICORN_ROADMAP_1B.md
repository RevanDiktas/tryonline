### **1. Body Measurement Extraction**

```python
# measurement_extraction.py

import cv2
import mediapipe as mp
import numpy as np
from typing import Dict

class BodyMeasurementExtractor:
    """
    Extracts body measurements from photos
    Uses MediaPipe Pose + custom ML models
    """
    
    def __init__(self):
        self.pose_detector = mp.solutions.pose.Pose()
        self.measurement_model = load_model('measurement_net_v3.h5')
    
    def extract_from_photos(
        self, 
        front_photo: np.ndarray, 
        side_photo: np.ndarray
    ) -> Dict[str, float]:
        """
        Extract measurements from front + side photos
        """
        # 1. Detect body landmarks
        front_landmarks = self.pose_detector.process(front_photo)
        side_landmarks = self.pose_detector.process(side_photo)
        
        # 2. Calculate reference scale (user's height or known object)
        pixel_to_cm_ratio = self.calculate_scale(front_landmarks)
        
        # 3. Extract key measurements
        measurements = {
            'height': self.calculate_height(front_landmarks, pixel_to_cm_ratio),
            'shoulders': self.calculate_shoulders(front_landmarks, pixel_to_cm_ratio),
            'bust': self.calculate_bust(front_landmarks, pixel_to_cm_ratio),
            'waist': self.calculate_waist(front_landmarks, side_landmarks, pixel_to_cm_ratio),
            'hips': self.calculate_hips(front_landmarks, pixel_to_cm_ratio),
            'inseam': self.calculate_inseam(front_landmarks, pixel_to_cm_ratio),
            'arm_length': self.calculate_arm_length(front_landmarks, pixel_to_cm_ratio),
        }
        
        # 4. ML refinement (learns from feedback)
        measurements = self.measurement_model.predict(measurements)
        
        # 5. Body type classification
        measurements['body_type'] = self.classify_body_type(measurements)
        
        # 6. Confidence scores
        measurements['confidence'] = self.calculate_confidence(front_landmarks, side_landmarks)
        
        return measurements

    def classify_body_type(self, measurements: Dict) -> str:
        """
        Classify into body types: Hourglass, Pear, Apple, Rectangle, Triangle
        """
        bust = measurements['bust']
        waist = measurements['waist']
        hips = measurements['hips']
        
        if abs(bust - hips) < 2 and waist < bust - 4:
            return 'hourglass'
        elif hips > bust + 2 and waist < hips - 4:
            return 'pear'
        elif bust > hips + 2:
            return 'triangle'
        elif abs(bust - waist) < 4:
            return 'rectangle'
        else:
            return 'apple'
```

    3. Similar users' purchase history
    4. Return data feedback
    """
    
    def __init__(self, database):
        self.db = database
        self.user_embeddings = self.load_user_embeddings()
        self.brand_profiles = self.load_brand_profiles()
        self.nn_model = NearestNeighbors(n_neighbors=100)
        self.nn_model.fit(self.user_embeddings)
    
    def recommend_size(
        self,
        user_id: str,
        brand_id: str,
        product_id: str
    ) -> Tuple[str, float]:
        """
        Returns: (recommended_size, confidence)
        """
        # 1. Get user measurements
        user_measurements = self.db.get_user_measurements(user_id)
        
        # 2. Get brand's size chart
        size_chart = self.db.get_brand_size_chart(brand_id, product_id)
        
        # 3. Direct measurement matching
        direct_match = self.match_to_size_chart(user_measurements, size_chart)
        
        # 4. Find similar users
        similar_users = self.find_similar_users(user_id, n=100)
        
        # 5. Check their purchase history with this brand
        similar_purchases = self.db.get_purchases(
            user_ids=similar_users,
            brand_id=brand_id,
            product_category=self.get_product_category(product_id)
        )
        
        # 6. Aggregate their sizes (what did they buy and keep?)
        size_distribution = self.aggregate_sizes(similar_purchases)
        
        # 7. Brand fit adjustment
        brand_fit = self.brand_profiles[brand_id]['fit_tendency']
        # 'runs_small', 'true_to_size', 'runs_large'
        
        # 8. Combine signals
        final_recommendation = self.ensemble_recommendation(
            direct_match=direct_match,
            collaborative_filtering=size_distribution,
            brand_fit=brand_fit,
            user_preferences=self.get_user_preferences(user_id)
        )
        
        # 9. Calculate confidence
        confidence = self.calculate_confidence(
            data_points=len(similar_purchases),
            agreement=self.check_signal_agreement(direct_match, size_distribution),
            brand_coverage=len(self.db.get_brand_purchases(brand_id))
        )
        
        return final_recommendation, confidence
    
    def learn_from_feedback(
        self,
        user_id: str,
        brand_id: str,
        product_id: str,
        recommended_size: str,
        actual_fit: str  # 'too_small', 'perfect', 'too_large'
    ):
        """
        Feedback loop: User reports fit, update models
        """
        # Update brand fit profile
        self.update_brand_profile(brand_id, product_id, recommended_size, actual_fit)
        
        # Update user preferences
        self.update_user_preferences(user_id, brand_id, actual_fit)
        
        # Retrain model periodically
        if self.should_retrain():
            self.retrain_model()
